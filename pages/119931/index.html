# 安装依赖

pnpm add ts-ui
1

# 使用

需要单独引入element样式和vxe-table样式 然后引入ts-ui自身的样式

// 使用elementUi
import elementUi from "element-ui";
Vue.use(elementUi);
// 使用tsui
import tsui from "ts-ui";
Vue.use(tsui);
// 引入样式
import "element-ui/lib/theme-chalk/index.css";
import "vxe-table/lib/style.css";
import "ts-ui/es/style/index.css";

# 使用组件

可直接通过'base-'加组件命名的方式使用

 <base-ComboBoxInput
    label="单位："
    placeholder="请选择单位"
    v-model="netId"
    fieldName="netId"
    :options="options"
/>
